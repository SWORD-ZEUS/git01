git config --global user.name "your_username" #配置用户信息
git config --global user.name your_email
git init #初始化代码管理仓库
git add file_name #添加到缓冲区
git reset HEAD #清空缓冲区
git status #查看当前状态，即工作区文件有没有添加到缓冲区
git commit -m ' 本次修改要说明的事项' #将缓冲区文件添加到本地仓库
git log #查看版本日志
git log -5 --pretty=oneline #显示五行，一次打印一行
git reset --hard HEAD^ #回退到上个版本
git reset --hard HEAD^^ #回退到上上个版本
git reset --hard HEAD~5 #回退5个版本
git reset --hard 唯一标识符的前5位 #去往指定的版本
git checkout git01.txt #恢复工作区误删除的git01.txt文件
git ls-files #查看本地库的文件
git checkout branch # 切换到指定分支
git checkout -b new_branch #新建分支并切换到新建分支
git branch -d branch #删除指定分支
git branch #查看所有分支,并且*号标记当前所在分支
git 2pranch #合并分支
git branch -m |-M #重命名分支，如果newbranch名字分支已经存在，则需要使用-M强制重命
